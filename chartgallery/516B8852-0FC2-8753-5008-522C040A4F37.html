3D 100% Stacked Area Chart SQL Script

-------------------------------------------------------------------------------------------------------------------------------------------------------------
-- 3D 100% Stacked Area Chart
--
--   Notes: The query below will work on the following database servers:
--          MS SQL Server, Sybase ASE, Sybase IQ, Sybase Anywhere, MySQL, PostreSQL
--
--   ORACLE: To make the query below work in Oracle, add " FROM DUAL" at the end of each SELECT         
--   DB2: To make the query below work in DB2, add " FROM SYSIBM.SYSDUMMY1" at the end of each SELECT
-------------------------------------------------------------------------------------------------------------------------------------------------------------

select 1 as category, 40 as s1, 75 as s2, 75 as s3, 75 as s4
union
select 2 as category, 5 as s1, 45 as s2, 75 as s3, 75 as s4
union
select 3 as category, 20 as s1, 80 as s2, 75 as s3, 75 as s4
union
select 4 as category, 70 as s1, 25 as s2, 75 as s3, 75 as s4
union
select 5 as category, 20 as s1, 75 as s2, 75 as s3, 75 as s4

select 6 as category, 5 as s1, 75 as s2, 75 as s3, 75 as s4

select 7 as category, 33 as s1, 60 as s2, 75 as s3, 75 as s4
union
select 8 as category, 95 as s1, 80 as s2, 75 as s3, 75 as s4
union
select 9 as category, 10 as s1, 50 as s2, 75 as s3, 75 as s4
union
select 10 as category, 70 as s1, 90 as s2, 75 as s3, 75 as s4
GO
